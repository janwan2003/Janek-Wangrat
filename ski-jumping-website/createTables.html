Stworzenie potrzebnych tabel

CREATE TABLE Zgloszenie(
data_zgloszenia DATE NOT NULL ,
id_zawodnika INTEGER NOT NULL REFERENCES Zawodnik
);
CREATE TABLE Zawodnik(
id INTEGER PRIMARY KEY,
imie VARCHAR(10) NOT NULL,
nazwisko VARCHAR(13) NOT NULL,
kraj VARCHAR(13) NOT NULL REFERENCES Firma
);
CREATE TABLE Skok(
id INTEGER PRIMARY KEY,
dlugosc INTEGER NOT NULL,
ocena INTEGER NOT NULL,
seria INTEGER NOT NULL,
id_zawodnika INTEGER NOT NULL REFERENCES Zawodnik
);
CREATE TABLE Kraj(
nazwa VARCHAR(13) PRIMARY KEY,
kwota_startowa INTEGER NOT NULL
);
CREATE TABLE Konkurs(
nazwa VARCHAR(30) PRIMARY KEY,
data_konkursu DATE NOT NULL,
termin_zgloszen DATE NOT NULL
);
